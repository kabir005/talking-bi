================================================================================
                         QUICK START GUIDE
================================================================================

STEP 1: GITHUB (2 minutes)
--------------------------------------------------------------------------------
cd talking-bi
git init
git add .
git commit -m "Initial commit"
git remote add origin https://github.com/YOUR_USERNAME/talking-bi.git
git push -u origin main

STEP 2: AWS EC2 (5 minutes)
--------------------------------------------------------------------------------
1. Go to AWS Console → EC2 → Launch Instance
2. Name: talking-bi-server
3. AMI: Ubuntu 22.04 LTS
4. Type: t2.medium
5. Key pair: Create/download .pem
6. Security: Allow ports 22, 80, 443, 8000
7. Storage: 20 GB
8. Launch!

STEP 3: DEPLOY (10 minutes)
--------------------------------------------------------------------------------
# Connect
ssh -i your-key.pem ubuntu@YOUR_EC2_IP

# Setup
curl -o setup.sh https://raw.githubusercontent.com/YOUR_USERNAME/talking-bi/main/deploy/aws-setup.sh
chmod +x setup.sh
./setup.sh

# Log out and back in
exit
ssh -i your-key.pem ubuntu@YOUR_EC2_IP

# Deploy
cd /opt
sudo git clone https://github.com/YOUR_USERNAME/talking-bi.git
sudo chown -R ubuntu:ubuntu talking-bi
cd talking-bi
cp .env.example .env
nano .env  # Add GROQ_API_KEY
chmod +x deploy/deploy.sh
./deploy/deploy.sh

DONE! Access at: http://YOUR_EC2_IP
